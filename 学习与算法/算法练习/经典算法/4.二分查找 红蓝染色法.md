class Solution(object):
    def findTheDistanceValue(self, arr1, arr2, d):
        """
        :type arr1: List[int]
        :type arr2: List[int]
        :type d: int
        :rtype: int
        """
        arr2.sort()
        ans = 0
        for x in arr1:
            i = bisect_left(arr2, x-d)

                ans += 1
        return ans
